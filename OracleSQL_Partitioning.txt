create table INTERNS_RANGE_DEP
(
  id NUMBER(6,0),
  name varchar2(30),
  department_id NUMBER(4,0)
)
PARTITION BY RANGE (department_id)
(
  PARTITION DEP_1_3 VALUES LESS THAN (4),
  PARTITION DEP_4_7 VALUES LESS THAN (7),
  PARTITION DEP_8_10 VALUES LESS THAN (11),
  PARTITION DEP_11 VALUES LESS THAN (100)
);
